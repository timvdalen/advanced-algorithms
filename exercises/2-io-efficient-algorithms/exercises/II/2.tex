Let $T(n)$ denote the time required to run \textsc{EM-MergeSort} on an array of length $n$.
\begin{equation}
\begin{split}
T(0) &= O(1)\\
T(1) &= O(1)\\
T(n) &= k * T(n/k) + O(k n)\\
&= \sum_{i = 0}^{\log_k{n}} (k^i * O(k * \frac{n}{k^i})) + O(n)\\
&= O((\log_k{n} + 1) *  k n)+ O(n)\\
&= O(k n \log_k{(k n)}) + O(n)\\
&= O(k n \log_k{(k n)})\\
\end{split}
\end{equation}
